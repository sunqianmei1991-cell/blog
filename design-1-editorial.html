class Agent:
    """ReAct Agent with tool use"""
    
    def __init__(self, llm, tools):
        self.llm = llm
        self.tools = tools
        
    def run(self, query):
        # Thought → Action → Observation
        while not self.is_done():
            thought = self.think(query)
            action = self.act(thought)
            obs = self.observe(action)
